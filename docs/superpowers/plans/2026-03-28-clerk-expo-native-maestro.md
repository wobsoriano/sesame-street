> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Maestro-based E2E tests for `@clerk/expo` native components (AuthView, UserButton) running on an Android emulator, integrated into the Clerk JS monorepo's existing CI infrastructure.

**Architecture:** New `expo-native` template with native Clerk components, Maestro YAML flows for sign-in testing, and conditional CI steps in the existing integration test matrix that boot an Android emulator and run Maestro instead of Playwright.

**Tech Stack:** Expo SDK 53, @clerk/expo (native), Maestro CLI, Android emulator (API 31), GitHub Actions with reactivecircus/android-emulator-runner

### Task 2: Create expo-native template - Metro config

**Files:**
- Create: `integration/templates/expo-native/metro.config.js`

- [ ] **Step 1: Create integration/templates/expo-native/metro.config.js**

Copy the same monorepo-aware Metro config from `integration/templates/expo-web/metro.config.js`. This file handles:
- Resolving `@clerk/*` packages from monorepo source
- Blocking duplicate React/React Native modules from pnpm store
- Custom subpath export resolution for `@clerk/react/internal` etc.

```js
/**
 * DO NOT EDIT THIS FILE UNLESS YOU DEFINITELY KNOW WHAT YOU ARE DOING.
 * THIS ENSURES THAT INTEGRATION TESTS ARE LOADING THE CORRECT DEPENDENCIES.
 */
const { getDefaultConfig } = require('expo/metro-config');
const packageJson = require('./package.json');
const path = require('node:path');

/** @type {() => string | undefined} */
const getClerkExpoPath = () => {
  const clerkExpoPath = packageJson.dependencies['@clerk/expo'];

  if (clerkExpoPath?.startsWith('*')) {
    const pathToModule = require.resolve('@clerk/expo');
    return pathToModule.replace('dist/index.js', '');
  }

  if (clerkExpoPath?.startsWith('file:')) {
    return clerkExpoPath.replace('file:', '');
  }

  if (clerkExpoPath?.startsWith('link:')) {
    return clerkExpoPath.replace('link:', '');
  }

  return undefined;
};

const clerkExpoPath = getClerkExpoPath();
const clerkMonorepoPath = clerkExpoPath?.replace(/\/packages\/expo$/, '');

/** @type {import('expo/metro-config').MetroConfig} */
const config = getDefaultConfig(__dirname);

// Only customize Metro config when running from monorepo
if (clerkMonorepoPath) {
  console.log('[Metro Config] Applying monorepo customizations');
  config.watchFolders = [clerkMonorepoPath];

  // Disable file watching to prevent infinite reload loops in integration tests
  config.watcher = {
    healthCheck: {
      enabled: false,
    },
  };

  // Prioritize local node_modules over monorepo node_modules
  config.resolver.nodeModulesPaths = [path.resolve(__dirname, 'node_modules'), `${clerkMonorepoPath}/node_modules`];

  // Explicitly map @clerk packages to their source locations
  config.resolver.extraNodeModules = {
    '@clerk/react': path.resolve(clerkMonorepoPath, 'packages/react'),
    '@clerk/expo': path.resolve(clerkMonorepoPath, 'packages/expo'),
    '@clerk/shared': path.resolve(clerkMonorepoPath, 'packages/shared'),
    '@clerk/types': path.resolve(clerkMonorepoPath, 'packages/types'),
  };

  // Block duplicate React/React Native from monorepo root and package node_modules
  const escapedPath = clerkMonorepoPath.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
  config.resolver.blockList = [
    new RegExp(`${escapedPath}/node_modules/react/`),
    new RegExp(`${escapedPath}/node_modules/react$`),
    new RegExp(`${escapedPath}/node_modules/react-dom/`),
    new RegExp(`${escapedPath}/node_modules/react-dom$`),
    new RegExp(`${escapedPath}/node_modules/react-native/`),
    new RegExp(`${escapedPath}/node_modules/react-native$`),
    new RegExp(`${escapedPath}/node_modules/\\.pnpm/.*/node_modules/react/`),
    new RegExp(`${escapedPath}/node_modules/\\.pnpm/.*/node_modules/react$`),
    new RegExp(`${escapedPath}/node_modules/\\.pnpm/.*/node_modules/react-dom/`),
    new RegExp(`${escapedPath}/node_modules/\\.pnpm/.*/node_modules/react-dom$`),
    new RegExp(`${escapedPath}/node_modules/\\.pnpm/.*/node_modules/react-native/`),
    new RegExp(`${escapedPath}/node_modules/\\.pnpm/.*/node_modules/react-native$`),
    new RegExp(`${escapedPath}/packages/.*/node_modules/react/`),
    new RegExp(`${escapedPath}/packages/.*/node_modules/react$`),
    new RegExp(`${escapedPath}/packages/.*/node_modules/react-dom/`),
    new RegExp(`${escapedPath}/packages/.*/node_modules/react-dom$`),
    new RegExp(`${escapedPath}/packages/.*/node_modules/react-native/`),
    new RegExp(`${escapedPath}/packages/.*/node_modules/react-native$`),
  ];

  // Custom resolver for @clerk package subpath exports
  const originalResolveRequest = config.resolver.resolveRequest;
  config.resolver.resolveRequest = (context, moduleName, platform) => {
    const clerkPackageMatch = moduleName.match(/^(@clerk\/[^/]+)\/(.+)$/);
    if (clerkPackageMatch && config.resolver.extraNodeModules) {
      const [, packageName, subpath] = clerkPackageMatch;
      const packageRoot = config.resolver.extraNodeModules[packageName];

      if (packageRoot) {
        const subpathDir = path.join(packageRoot, subpath);
        try {
          const subpathPkg = require(path.join(subpathDir, 'package.json'));
          if (subpathPkg.main) {
            const resolvedPath = path.join(subpathDir, subpathPkg.main);
            return { type: 'sourceFile', filePath: resolvedPath };
          }
        } catch (e) {
          // Subpath directory doesn't exist, continue with default resolution
        }
      }
    }

    if (originalResolveRequest) {
      return originalResolveRequest(context, moduleName, platform);
    }
    return context.resolveRequest(context, moduleName, platform);
  };
}

module.exports = config;
```

```bash
#!/bin/bash
set -e

APP_ID="${APP_ID:?APP_ID env var required}"
TEST_EMAIL="${TEST_EMAIL:?TEST_EMAIL env var required}"
TEST_PASSWORD="${TEST_PASSWORD:?TEST_PASSWORD env var required}"
MAX_RETRIES=3
DELAYS=(0 20 60)

for i in $(seq 1 $MAX_RETRIES); do
  echo "=== Attempt $i of $MAX_RETRIES ==="

  if [ "${DELAYS[$i-1]}" -gt 0 ]; then
    echo "Waiting ${DELAYS[$i-1]}s before retry..."
    sleep "${DELAYS[$i-1]}"
  fi

  set +e
  maestro test integration/tests/expo-native/.maestro/ \
    --env=APP_ID="$APP_ID" \
    --env=TEST_EMAIL="$TEST_EMAIL" \
    --env=TEST_PASSWORD="$TEST_PASSWORD" \
    --format=junit \
    --output "report_attempt_${i}.xml"
  EXIT_CODE=$?
  set -e

  adb shell screencap -p "/sdcard/screenshot_attempt_${i}.png"
  adb pull "/sdcard/screenshot_attempt_${i}.png" "screenshot_attempt_${i}.png" 2>/dev/null || true

  if [ $EXIT_CODE -eq 0 ]; then
    echo "=== Tests passed on attempt $i ==="
    exit 0
  fi

  echo "=== Attempt $i failed (exit code $EXIT_CODE) ==="
done

echo "=== All $MAX_RETRIES attempts failed ==="
exit 1
```

- [ ] **Step 3: Create the build script at integration/tests/expo-native/build.mjs**

This script sets up the template app, writes env vars, runs prebuild and Gradle build:

```js
import { exec as execCb } from 'node:child_process';
import { promisify } from 'node:util';
import { resolve } from 'node:path';
import { existsSync, mkdirSync, writeFileSync, cpSync } from 'node:fs';

const exec = promisify(execCb);

const TEMPLATE_DIR = resolve(import.meta.dirname, '../../templates/expo-native');
const BUILD_DIR = resolve(import.meta.dirname, '../../../.tmp-expo-native');

async function run(cmd, opts = {}) {
  console.log(`> ${cmd}`);
  const { stdout, stderr } = await exec(cmd, { cwd: BUILD_DIR, ...opts });
  if (stdout) console.log(stdout);
  if (stderr) console.error(stderr);
}

async function main() {
  // Copy template to temp build dir
  if (existsSync(BUILD_DIR)) {
    await exec(`rm -rf ${BUILD_DIR}`);
  }
  mkdirSync(BUILD_DIR, { recursive: true });
  cpSync(TEMPLATE_DIR, BUILD_DIR, { recursive: true });

  // Write .env with Clerk keys from INTEGRATION_INSTANCE_KEYS
  const instanceKeys = process.env.INTEGRATION_INSTANCE_KEYS;
  if (instanceKeys) {
    const keys = JSON.parse(instanceKeys);
    const emailCodesKeys = keys['with-email-codes'];
    if (emailCodesKeys) {
      writeFileSync(
        resolve(BUILD_DIR, '.env'),
        `EXPO_PUBLIC_CLERK_PUBLISHABLE_KEY=${emailCodesKeys.pk}\n`
      );
    }
  }

  // Install dependencies (pkglab handles @clerk/expo)
  await run('pnpm install');

  // Prebuild Android
  await run('npx expo prebuild --platform android');

  // Build release APK
  await run('./gradlew assembleRelease', { cwd: resolve(BUILD_DIR, 'android') });

  const apkPath = resolve(BUILD_DIR, 'android/app/build/outputs/apk/release/app-release.apk');
  console.log(`APK built at: ${apkPath}`);

  // Write APK path to a file the CI step can read
  writeFileSync(resolve(BUILD_DIR, 'apk-path.txt'), apkPath);
}

main().catch(err => {
  console.error(err);
  process.exit(1);
});
```
